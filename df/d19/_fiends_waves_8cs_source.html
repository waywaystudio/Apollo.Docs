48 public void SingleSpawn(Action<Fiend> onSpawned)
49 {
50 if (IsSpawnEnd || totalFiendsCount == 0) return;
51
52 spawnTransform = spawnTransformList.RandomPick();
53 currentWave.RemainCount--;
54
55 ChallengeDirector.Character
56 .SpawnFiend(currentWave.FiendCode, spawnTransform.position, fiend =>
57 {
58 CallNextGroupFiend(fiend);
59 onSpawned?.Invoke(fiend);

61 ;

82 private void SetCurrentWave()
83 {
84 if (currentWave.RemainCount > 0) return;

86 // It was Last Fiend
87 if (currentWave == spawnEntityList.LastOrDefault())

99 private async UniTask SerialSpawnTask(Action<Fiend> onSpawned)

115 private void OnFiendDead()
116 {
117 totalFiendsCount--;

119 if (totalFiendsCount <= 0)
120 {
121 IsAllDead = true;
122 combatSection.TryCompletion();

124
125 if (NextGroupEntry is null) return;
126
127 switch (NextGroupEntry.SpawningType)
128 {
129 case FiendsSpawnType.Relay:
130 {
131 // Next Fiends Count is larger than current Group's Fiends Count;
132 if (totalFiendsCount <= 0)
133 {
134 NextGroupEntry.SerialSpawn(TellWhereIsDred);
135 }
136 else
137 {
138 NextGroupEntry.SingleSpawn(TellWhereIsDred);
139 }
140
141 break;
142 }
143 case FiendsSpawnType.Group:
144 {
145 if (totalFiendsCount <= 0)
146 {
147 NextGroupEntry.SerialSpawn(TellWhereIsDred);
148 }
149
150 break;
151 }
152 default: return;
153 }
