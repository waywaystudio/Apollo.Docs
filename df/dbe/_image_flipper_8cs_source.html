68 public void ModifyVertices(List<UIVertex> verts)
69 {
70 if (!enabled)
71 return;
72
73 var rt = transform as RectTransform;
74
75 for (var i = 0; i < verts.Count; ++i)
76 {
77 var v = verts[i];
78
79 // Modify positions
80 v.position = new Vector3(
81 (horizontal ? (v.position.x + (rt.rect.center.x - v.position.x) * 2) : v.position.x),
82 (vertical ? (v.position.y + (rt.rect.center.y - v.position.y) * 2) : v.position.y),
83 v.position.z
84 );
85
86 // Apply
87 verts[i] = v;
88 }
