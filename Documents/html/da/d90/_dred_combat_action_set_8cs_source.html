125 private void OnSkillSetChanged(Skill newSkill)
126 {
127 var assignedActionType = WeaponActionSlotType.None;
128
129 if (newSkill == MainAttackSkill) assignedActionType = WeaponActionSlotType.Weapon1Attack;
130 else if (newSkill == MainSpecialSkill) assignedActionType = WeaponActionSlotType.Weapon1Special;
131 else if (newSkill == SubAttackSkill) assignedActionType = WeaponActionSlotType.Weapon2Attack;
132 else if (newSkill == SubSpecialSkill) assignedActionType = WeaponActionSlotType.Weapon2Special;
133 else if (newSkill == UltimateSkill) assignedActionType = WeaponActionSlotType.Ultimate;
134
135 if (assignedActionType != WeaponActionSlotType.None)
136 {
137 OnSkillChangedTable[assignedActionType].Invoke(newSkill);

150 private void TryInvokeSkill(ISkillBehaviour skill)
151 {
152 if (CurrentSkillBehaviour.IsNullOrDestroyed())
153 {
154 skill.Active();
155 return;
156 }
157
158 if (isReserved) return;
159
160 reservationTask = new CancellationTokenSource();
161
162 ReserveNextAction(skill, reservationTask).Forget();
163 }
164
165 private async UniTaskVoid ReserveNextAction(ISkillBehaviour skill, CancellationTokenSource cts)
166 {
167 isReserved = true;
168
169 var tryDuration = reservePermitDuration;
170
171 while (tryDuration > 0f)
172 {
173 tryDuration -= Time.deltaTime;
174
175 if (skill.Invoker.IsAbleToActive)
176 {
177 skill.Active();
178 isReserved = false;
179
180 return;
181 }
182
183 await UniTask.Yield(cts.Token);
184 }
185
186 isReserved = false;
187 }

189 private void OnDestroy()
