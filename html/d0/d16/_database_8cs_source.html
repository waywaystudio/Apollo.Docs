93 // SheetData Generator
94 private void CreateAndUpdateDataObjects()
95 {
96 Finder.TryGetObjectList(dataScriptPath, $"t:MonoScript, Data", out List<MonoScript> monoList);
97
98 monoList.ForEach(x =>
99 {
100 if (!x.name.EndsWith("Data")) return;
101 if (!Finder.TryGetObject(dataObjectPath, x.name, out ScriptableObject dataObject))
102 {
103 dataObject = Finder.CreateScriptableObject(dataObjectPath, x.name, x.name);
104 }
105
106 var dataObjectType = dataObject.GetType();
107 var info = dataObjectType.GetMethod("LoadFromJson", BindingFlags.NonPublic | BindingFlags.Instance);
108
109 if (info != null)
110 {
111 info.Invoke(dataObject, null);
112 }
113
114 EditorUtility.SetDirty(dataObject);
115 });
116
117 Finder.TryGetObjectList(out sheetDataList);
118
119 sheetDataList.ForEach(x => x.Index = x.name.Replace("Data", "").ToEnum<DataIndex>());
120 sheetDataList.Sort((dataA, dataB) => dataA.Index.CompareTo(dataB.Index));
121
122 AssetDatabase.Refresh();
123 }
