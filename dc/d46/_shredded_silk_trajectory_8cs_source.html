53 private Vector3[] GenerateFlutterPath(Vector3 destination, int frequency, float strength)
54 {

57 path[0] = transform.position;

64 var linearPointEnd = Vector3.Lerp(position, destination, ratio + 1f / frequency);
65 var midPoint = Vector3.Lerp(linearPointStart, linearPointEnd, 0.5f);

67 midPoint += Vector3.up * Random.Range(1f, strength);

69 path[i * 2 + 1] = midPoint;

71 }
