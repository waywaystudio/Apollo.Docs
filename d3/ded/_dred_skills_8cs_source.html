53 public bool TryGetSkillBehaviour(DataIndex skillCode, out SkillBehaviour skillBehaviour)
54 {
55 skillBehaviour = null;
56
57 var weaponType = (WeaponType)skillCode.GetSecondBar();
58 var skillType = (SkillType)skillCode.GetThirdBar();
59
60 if (!weaponTable.ContainsKey(weaponType)) return false;
61 if (weaponTable[weaponType][skillType].IsNullOrDestroyed()) return false;
62
63 skillBehaviour = weaponTable[weaponType][skillType][skillCode]?.MainBehaviour;
64
65 return skillBehaviour.HasComponent();
66 }
