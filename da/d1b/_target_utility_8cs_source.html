73 public static List<T> GetTargetsInRect<T>(Vector3 pivot, Vector3 forward, LayerMask layer, float width, float height, Collider[] buffer)
74 {
75 Array.Clear(buffer, 0, buffer.Length);
76
77 var center = pivot + forward * (height * 0.5f);
78 var halfExtents = new Vector3(width * 0.5f, height * 0.5f, height * 0.5f);
79
80 // Rotate the forward vector 90 degrees around the Y axis for the "right" direction.
81 var right = Quaternion.Euler(0, 90, 0) * forward;
82 var orientation = Quaternion.LookRotation(forward, Vector3.up);
83 var size = Physics.OverlapBoxNonAlloc(center, halfExtents, buffer, orientation, layer);
84 var result = new List<T>();
85
86 for (var i = 0; i < size; i++)
87 {
88 var collider = buffer[i];
89
90 if (collider is null || !collider.TryGetComponent(out T taker)) continue;
91
92 var projectedPosition = Vector3.ProjectOnPlane(collider.transform.position - pivot, Vector3.up);
93 var toProjectedPosition = projectedPosition - Vector3.zero;
94
95 if (Vector3.Dot(toProjectedPosition, forward) <= height &&
96 Vector3.Dot(toProjectedPosition, right) <= width * 0.5f)
97
98 result.Add(taker);
99 }
100
101 return result;

129 public static List<T> GetTakersInWidthRaycast<T>(Vector3 pivot, Vector3 forward, RaycastHit[] buffer, LayerMask layer, float width, float height, int pierceCount)
130 {
131 // Modify pivot's Y value
132 pivot.y += 1f;
133
134 // Create a halfExtents vector with half of width and height
135 var halfExtents = new Vector3(width / 2f, height / 2f, 0f);
136
137 // Set up the orientation of the box
138 var orientation = Quaternion.LookRotation(forward);
139
140 // Do the BoxCast
141 var hitCount = Physics.BoxCastNonAlloc(pivot, halfExtents, forward, buffer, orientation, Mathf.Infinity, layer);
142
143 var takers = new List<T>();
144 var hitDistanceMap = new List<KeyValuePair<T, float>>();
145 for (var i = 0; i < hitCount; i++)
146 {
147 var hit = buffer[i];
148 if (hit.collider.TryGetComponent(out T taker))
149 {
150 // ICombatTaker의 경우 Alive 여부 판단 필요.
151 hitDistanceMap.Add(new KeyValuePair<T, float>(taker, hit.distance));
152 }
153 }
154
155 // Clear unused entries in buffer
156 for (var i = hitCount; i < buffer.Length; i++) buffer[i] = default;
157
158 hitDistanceMap.Sort((a, b) => a.Value.CompareTo(b.Value));
159
160 // If pierceCount is 0, return only the nearest taker
161 if (pierceCount == 0 && hitDistanceMap.Count > 0)
162 takers.Add(hitDistanceMap[0].Key);
163 else
164 // Return the 'pierceCount' number of nearest takers
165 for (var i = 0; i < pierceCount && i < hitDistanceMap.Count; i++)
166 takers.Add(hitDistanceMap[i].Key);
167
168 return takers;
169 }
