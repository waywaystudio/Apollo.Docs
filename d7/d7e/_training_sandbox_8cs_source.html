57 public void SpawnDummy()
58 {
59 var battleSequencePositions = new List<Vector3>();
60
61 if (BattleSequence is not null)
62 {
63 BattleSequence.PositionList.ForEach(tr => battleSequencePositions.Add(tr.position));
64 }
65
66 var tempPositions = BattleSequence is null
67 ? RandomPositions(16)
68 : battleSequencePositions;
69
70
71 spawnEntityList.ForEach(wave =>
72 {
73 for (var i = 0; i < wave.Count; i++)
74 {
75 if (tempPositions.IsNullOrEmpty())
76 {
77 ResetTempPositions();
78 }
79
80 var spawnPosition = tempPositions.RandomPick();
81
82 ChallengeDirector.Character.SpawnFiend(wave.FiendCode, spawnPosition);
83 }
84 });
85
86 return;
87
88 void ResetTempPositions()
89 {
90 tempPositions = BattleSequence is null
91 ? RandomPositions(16)
92 : battleSequencePositions;
93 }
94 }

96 private List<Vector3> RandomPositions(int count)
97 {
98 if (Dred.IsNullOrDestroyed()) return null;
99
100 var result = new List<Vector3>(count);
101
102 for (var i = 0; i < count; i++)
103 {
104 var randomPosition = Dred.Position + Random.insideUnitCircle.ToVector3XZ() * 10f;
105
106 result.Add(randomPosition);
107 }
108
109 return result;
110 }

420 {
421 await UniTask.WaitUntil(() => ChallengeDirector.IsStartChallenge).TimeoutWithoutException(TimeSpan.FromSeconds(5));
422
423 // TODO. If want to test any Stage (Battle, boss and so on), delete below 2 lines.
424 var csd = GetComponentInParent<ChallengeDungeonDirector>();
425 if (csd.NextStageRoomType != StageRoomType.Training) return;
426
427 if (spawnWithStart) SpawnDummy();
428 if (spawnEndless)
