87 public List<GameObject> GetObjects(DetectProperty property)
88 {
89 var pivot = master.position + property.Offset;
90 var forward = master.forward;
91 var detectedObjects = property.TargetType switch
92 {
93 TargetingType.Self => new List<GameObject> { master.gameObject },
94 TargetingType.Circle => InSphere(pivot, forward, property.TargetLayer, property.SizeEntity.Y),
95 TargetingType.Arc => InAngle(pivot, forward, property.TargetLayer, property.SizeEntity.Y, property.SizeEntity.Z),
96 TargetingType.Rect => InRect(pivot, forward, property.TargetLayer, property.SizeEntity.Z, property.SizeEntity.Y),
97 TargetingType.Donut => InDonut(pivot, property.TargetLayer, property.SizeEntity),
98 TargetingType.Raycast => CastRay(pivot, forward, property.TargetLayer, property.SizeEntity.Y),
99 TargetingType.Line => CastRay(pivot, forward, property.TargetLayer, property.SizeEntity.Y),
100 TargetingType.Colliding => null,
101 _ => null,
102 };
103
104 return detectedObjects;

110 public List<GameObject> InAngle(Vector3 center, Vector3 forward, LayerMask layerMask, float radius, float angle)
111 {
112 Array.Clear(ColliderBuffer, 0, ColliderBuffer.Length);
113
114 if (Physics.OverlapSphereNonAlloc(center, radius, ColliderBuffer, layerMask) == 0) return null;
115
116 ObjectList.Clear();
117 ColliderBuffer.ForEach(collider =>
118 {
119 if (collider.IsNullOrEmpty()) return;
120
121 if (!angle.Approximately(360.0f))
122 {
123 var direction = (collider.transform.position - center).normalized;
124
125 if (Vector3.Angle(forward, direction) > angle * 0.5f) return;
126 }
127
128 ObjectList.Add(collider.gameObject);
129 });
130
131 return ObjectList;
132 }

134 public List<GameObject> InRect(Vector3 pivot, Vector3 forward, LayerMask layer, float width, float height)
135 {
136 Array.Clear(ColliderBuffer, 0, ColliderBuffer.Length);
137
138 var center = pivot + forward * (height * 0.5f);
139 var halfExtents = new Vector3(width * 0.5f, height * 0.5f, height * 0.5f);
140
141 // Rotate the forward vector 90 degrees around the Y axis for the "right" direction.
142 var right = Quaternion.Euler(0, 90, 0) * forward;
143 var orientation = Quaternion.LookRotation(forward, Vector3.up);
144 var size = Physics.OverlapBoxNonAlloc(center, halfExtents, ColliderBuffer, orientation, layer);
145
146 ObjectList.Clear();
147
148 for (var i = 0; i < size; i++)
149 {
150 var colliderObject = ColliderBuffer[i];
151
152 if (colliderObject is null) continue;
153
154 var projectedPosition = Vector3.ProjectOnPlane(colliderObject.transform.position - pivot, Vector3.up);
155 var toProjectedPosition = projectedPosition - Vector3.zero;
156
157 if (Vector3.Dot(toProjectedPosition, forward) <= height &&
158 Vector3.Dot(toProjectedPosition, right) <= width * 0.5f)
159 ObjectList.Add(colliderObject.gameObject);
160 }
161
162 return ObjectList;
163 }

165 public List<GameObject> InDonut(Vector3 pivot, LayerMask layerMask, SizeEntity entity)
166 {
167 var outerRadius = entity.Z;
168 var innerRadius = entity.Y;
169
170 // Get all targets within the larger sphere.
171 var outerCount = Physics.OverlapSphereNonAlloc(pivot, outerRadius, ColliderBuffer, layerMask);
172 if (outerCount == 0) return null;
173
174 // Get all targets within the smaller sphere.
175 var innerCount = Physics.OverlapSphereNonAlloc(pivot, innerRadius, SubColliderBuffer, layerMask);
176
177 // Reset the taker list.
178 ObjectList.Clear();
179
180 // Go through all targets in the outer sphere.
181 for (var i = 0; i < outerCount; i++)
182 {
183 var outerCollider = ColliderBuffer[i];
184
185 // Skip null colliders.
186 if (outerCollider is null) continue;
187
188 // If this target is also in the inner sphere, skip it.
189 if (Array.IndexOf(SubColliderBuffer, outerCollider, 0, innerCount) >= 0) continue;
190
191 ObjectList.Add(outerCollider.gameObject);
192 }
193
194 return ObjectList;
195 }

197 public List<GameObject> InLine(Vector3 pivot, Vector3 forward, LayerMask layer, float width, float height)
198 {
199 Array.Clear(RayBuffer, 0, RayBuffer.Length);
200
201 // Create a halfExtents vector with half of width and height
202 var halfExtents = new Vector3(width * 0.5f, height * 0.5f, 0f);
203 var orientation = Quaternion.LookRotation(forward);
204 var hitCount = Physics.BoxCastNonAlloc(pivot, halfExtents, forward, RayBuffer, orientation, Mathf.Infinity, layer);
205
206 if (hitCount == 0) return null;
207
208 ObjectList.Clear();
209
210 for (var i = 0; i < hitCount; i++)
211 {
212 var hit = RayBuffer[i];
213
214 ObjectList.Add(hit.collider.gameObject);
215 }
216
217 ObjectList.Sort(pivot, SortingType.DistanceAscending);
218
219 return ObjectList;
220 }
