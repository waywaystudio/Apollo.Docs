66 private List<GoodWeightProperty> WeightedRandomSelection(IEnumerable<GoodWeightProperty> goods, int count)
67 {
68 var selectedGoods = new List<GoodWeightProperty>();
69 var availableGoods = new List<GoodWeightProperty>(goods);
70
71 for (var i = 0; i < count && availableGoods.Count > 0; i++)
72 {
73 float totalWeight = 0;
74 foreach (var good in availableGoods)
75 {
76 totalWeight += good.Weight;
77 }
78
79 var randomPoint = UnityEngine.Random.value * totalWeight;
80 var currentPoint = 0f;
81
82 for (var j = 0; j < availableGoods.Count; j++)
83 {
84 currentPoint += availableGoods[j].Weight;
85
86 if (currentPoint >= randomPoint)
87 {
88 selectedGoods.Add(availableGoods[j]);
89 availableGoods.RemoveAt(j);
90 break;
91 }
92 }
93 }
94
95 return selectedGoods;
