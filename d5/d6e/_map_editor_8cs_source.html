120 private bool TryReadCsv(out CSVMatrix matrix)
121 {
122 matrix = new CSVMatrix();
123
124 if (csvFile is null)
125 {
126 Debug.LogError("No csv file to parse");
127 return false;
128 }
129
130 var rows = csvFile.text.Split(new[] {'\n'});
131
132 for (var i = 0; i < rows.Length; i++)
133 {
134 var entities = rows[i].Split(new[] {','});
135
136 matrix.Add(entities.ToList());
137 }
138
139 return matrix.HasElement();
140 }

147 private void CreateNewTerrain()
148 {
149 if (!Verify.IsTrue(TryReadCsv(out matrix))) return;
150 if (!Verify.IsTrue(AssetDatabase.IsValidFolder(terrainDataPath))) return;
151 if (CurrentMap.IsNullOrDestroyed()) return;
152
153 AssetDatabase.CopyAsset(terrainDataPath + $"/{terrainDataTemplate.name}.asset", terrainDataPath + $"/TerrainData.{MapName}.asset");
154 Finder.TryGetObject(terrainDataPath, $"TerrainData.{MapName}", out TerrainData newTerrainData);
155
156 var terrain = CurrentMap.GetComponentInChildren<Terrain>();
157 var terrainCollider = CurrentMap.GetComponentInChildren<TerrainCollider>();
158 var terrainWidth = matrix.FirstOrDefault().Count * gridSize + terrainOffset * 2;
159 var terrainLength = matrix.Count * gridSize + terrainOffset * 2;
160 var terrainObjectTransform = terrain.transform;
161
162 terrain.terrainData = newTerrainData;
163 terrainCollider.terrainData = newTerrainData;
164 newTerrainData.size = new Vector3(terrainWidth, newTerrainData.size.y, terrainLength);
165 terrainObjectTransform.position = new Vector3(-terrainWidth / 2, terrainObjectTransform.position.y, -terrainLength / 2);
166
167 /*
168 * Water
169 */
170 Water.localScale = new Vector3(terrainWidth / 10f, 1, terrainLength / 5f);
