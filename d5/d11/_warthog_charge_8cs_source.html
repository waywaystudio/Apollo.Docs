59 private async UniTaskVoid TargetDashTask(CancellationTokenSource cts)
60 {
61 while (!cts.IsCancellationRequested)
62 {
63 if (SkillTarget.IsNullOrDestroyed())
64 {
65 return;
66 }
67
68 // Move Forward

70 var mainCharacterMoveSpeed = Character.StatTable[StatType.MoveSpeed];

72 mainCharacterTransform.Translate(mainCharacterTransform.forward * (dashSpeedMultiplier * mainCharacterMoveSpeed * Time.fixedDeltaTime), Space.World);

74 // Rotate
