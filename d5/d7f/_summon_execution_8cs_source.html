51 private async UniTaskVoid SingleSpawn()
52 {
53 if (IsSpawnEnd || TotalFiendsCount == 0) return;
54
55 var randomPosition = SpawnPositionList.RandomPick();
56 var fiendPrefab = Database.FiendPrefabs[CurrentWave.FiendCode];
57 var spawnParticle = spawnEffectPool.Get();
58
59 CurrentWave.RemainCount--;
60
61 spawnParticle.transform.SetPositionAndRotation(randomPosition, spawnParticle.transform.rotation);
62 spawnParticle.transform.SetParent(null);
63 spawnParticle.Play();
64
65 await UniTask.Delay(500, false, PlayerLoopTiming.FixedUpdate, spawnParticle.GetCancellationTokenOnDestroy());
66
67 var fiendObject = Instantiate(fiendPrefab, randomPosition, Quaternion.identity, null);
68 if (!fiendObject.TryGetComponent(out Fiend fiend)) return;
69
70 fiend.Initialize();
71 SetCurrentWave();
72 }

74 private async UniTask SerialSpawnTask()

80 await UniTask.Delay((int)(serialInterval * 1000f),
