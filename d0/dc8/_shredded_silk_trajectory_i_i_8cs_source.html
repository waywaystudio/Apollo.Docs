33 private async UniTaskVoid FlutterTowardsTarget(CancellationTokenSource cts)
34 {
35 var strengthPitch = Random.Range(flutterStrength * 0.8f, flutterStrength * 1.2f);
36 var frequencyPitch = Random.Range(flutterFrequency - 1, flutterFrequency + 1);
37 var durationPitch = Random.Range(duration * 0.8f, duration * 1.2f);
38 var path = GenerateFlutterPath(Destination, frequencyPitch, strengthPitch);
39 var timeElapsed = 0f;
40
41 // DOTween을 사용하여 경로를 따라 오브젝트를 이동시킵니다.
42 Projectile.transform.DOPath(path, durationPitch, PathType.CatmullRom)
43 .SetOptions(false)
44 .SetEase(flutterEase)
45 .OnComplete(() => Projectile.Invoker.Complete());
46
47 // UniTask를 사용하여 비동기적으로 대기합니다.
48 while (!cts.IsCancellationRequested && timeElapsed < duration)
49 {
50 timeElapsed += Time.deltaTime;
51
52 await UniTask.Yield(PlayerLoopTiming.Update, cts.Token);
53 }

56 // 목표 위치까지 나풀거리며 이동할 경로를 생성합니다.
57 private Vector3[] GenerateFlutterPath(Vector3 destination, int frequency, float strength)
58 {
59 var path = new Vector3[(frequency + 1) * 2]; // 각 지점 사이에 중간 지점을 추가하기 위해 배열 크기 조정
60 path[0] = transform.position; // 시작 위치
61
62 for (var i = 0; i < frequency; i++)
63 {
64 var ratio = (float)i / frequency;
65 var position = transform.position;
66 var linearPointStart = Vector3.Lerp(position, destination, ratio);
67 var linearPointEnd = Vector3.Lerp(position, destination, ratio + 1f / frequency);
68
69 // 시작점과 끝점 사이의 중간 지점을 계산합니다.
70 var midPoint = Vector3.Lerp(linearPointStart, linearPointEnd, 0.5f);
71 // 중간 지점의 높이를 증가시켜 포물선의 볼록한 부분을 만듭니다.
72 midPoint += Vector3.up * Random.Range(1f, strength);
73
74 // 경로 배열에 시작점, 중간 포물선 지점, 끝점을 추가합니다.
75 path[i * 2 + 1] = midPoint;
76 path[i * 2 + 2] = linearPointEnd;
77 }
78
79 path[^1] = destination; // 목표 위치
80 return path;
