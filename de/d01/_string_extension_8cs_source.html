26 public static string ToPascalCase(this string original)
27 {
28 var invalidCharsRgx = new Regex("[^_a-zA-Z0-9]");
29 var whiteSpace = new Regex(@"(?<=\s)");
30 var startsWithLowerCaseChar = new Regex("^[a-z]");
31 var firstCharFollowedByUpperCasesOnly = new Regex("(?<=[A-Z])[A-Z0-9]+$");
32 var lowerCaseNextToNumber = new Regex("(?<=[0-9])[a-z]");
33 var upperCaseInside = new Regex("(?<=[A-Z])[A-Z]+?((?=[A-Z][a-z])|(?=[0-9]))");
34
35 // replace white spaces with underscore, then replace all invalid chars with empty string
36 var pascalCase = invalidCharsRgx.Replace(whiteSpace.Replace(original, "_"), string.Empty)
37 .Split(new [] { '_' }, StringSplitOptions.RemoveEmptyEntries)
38 .Select(w => startsWithLowerCaseChar.Replace(w, m => m.Value.ToUpper()))
39 .Select(w => firstCharFollowedByUpperCasesOnly.Replace(w, m => m.Value.ToLower()))
40 .Select(w => lowerCaseNextToNumber.Replace(w, m => m.Value.ToUpper()))
41 .Select(w => upperCaseInside.Replace(w, m => m.Value.ToLower()));
42
43 var result = string.Concat(pascalCase);
44
45 return result.Length <= 2
46 ? result.ToUpper()
47 : result;
48 }
