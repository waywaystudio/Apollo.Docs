48 private void AddIgnitedForceToProjectile(Snare snare)
49 {
50 if (snare is not ThreadedSanctumSnare sanctumSnare) return;
51
52 // Active ThreadedSanctum's Conflagration Vfx
53 if (snare.Effector.TryGetEffect(out ParticleEffect conflagrationFx, "Vfx:ConflagrationLoop"))
54 {
55 conflagrationFx.Activation = true;
56 }
57
58 sanctumSnare.Trigger.AddOnEnter("EnforceSilkStringsProjectile", dredObject =>
59 {
60 if (!dredObject.TryGetComponent(out Dred _)) return;
61 if (delayTask is not null)
62 {
63 delayTask.Cancel();
64 }
65
66 projectileSpawnEvent.Add(DataIndex.SilkTwisterProjectile, "AddIgnitedEffect", AddIgnitedEffect);
67
68 // Active SilkTwisterProjectile's Ignited Effect
69 silkTwister.ActivatedProjectiles?.ForEach(projectile =>
70 {
71 if (projectile.Effector.TryGetEffect<ParticleEffect>(out var ignitedVfx, "Vfx:SilkIgnited"))
72 {
73 ignitedVfx.Activation = true;
74 ignitedVfx.Play();
75 }
76 });
77 });
78
79 sanctumSnare.Trigger.AddOnExit("ResetSilkStringsProjectile", dredObject =>
80 {
81 if (!dredObject.TryGetComponent(out Dred _)) return;
82 delayTask = new CancellationTokenSource();
83
84 DelayedRemoveEffect(snare, delayTask).Forget();
85 });
86 }
87
88 private void RemoveIgnitedForceFromProjectile(Snare snare)
89 {
90 if (snare is not ThreadedSanctumSnare sanctumSnare) return;
91
92 // @SH : Active ThreadedSanctum's Conflagration Vfx
93 if (snare.Effector.TryGetEffect(out ParticleEffect conflagrationFx, "Vfx:ConflagrationLoop"))
94 {
95 conflagrationFx.Activation = false;
96 conflagrationFx.Stop();
97 }
98
99 sanctumSnare.Trigger
100 .RemoveOnEnter("EnforceSilkStringsProjectile")
101 .RemoveOnExit("ResetSilkStringsProjectile")
102 ;
103 }

105 // ProjectileSpawnEvent를 통해서, SilkStringProjectile.Executor에 Ignited StatusEffect를 enable.
106 // 좀 복잡해지는 것 같아서, SilkStringsProjectile에 미리 StatusEff.ectExecutor를 붙여놓고, 키고 끄는 것으로 바꿈.
107 private static void AddIgnitedEffect(Projectile projectile)
108 {
109 projectile.Sections
110 .Add(Section.Active, "PlaySilkIgnitedEffect", () =>
111 {
112 if (projectile.Effector.TryGetEffect<ParticleEffect>(out var vfx, "Vfx:SilkIgnited"))
113 {
114 vfx.Activation = true;
115 vfx.Play();
116 }
117 });
118
119 if (projectile.Effector.TryGetEffect<ParticleEffect>(out var ignitedVfx, "Vfx:SilkIgnited"))
120 {
121 ignitedVfx.Activation = true;
122 ignitedVfx.Play();
123 }
124
125 if (!projectile.Executor.TryGetExecutions<StatusEffectExecutor>(out var statusEffectExe)) return;
126
127 statusEffectExe.ForEach(exe => exe.Activation = true);
128 }

130 private async UniTaskVoid DelayedRemoveEffect(Snare snare, CancellationTokenSource cts)
131 {
132 await UniTask.Delay(TimeSpan.FromSeconds(quality.Value), DelayType.DeltaTime, cancellationToken:cts.Token);
133
134 projectileSpawnEvent.Remove(DataIndex.SilkTwisterProjectile, "AddIgnitedEffect", RemoveIgnitedEffect);
135 delayTask = null;
136 }

138 private static void RemoveIgnitedEffect(Projectile projectile)
139 {
140 projectile.Sections.Remove(Section.Active, "PlaySilkIgnitedEffect");
141
142 // @SH : DeActive Projectile's Ignited Effect
143 if (projectile.Effector.TryGetEffect<ParticleEffect>(out var ignitedVfx, "Vfx:SilkIgnited"))
144 {
145 ignitedVfx.Activation = false;
146 ignitedVfx.Stop();
147 }
148
149 if (!projectile.Executor.TryGetExecutions<StatusEffectExecutor>(out var statusEffectExe)) return;
150
151 statusEffectExe.ForEach(exe => exe.Activation = false);
152 }
