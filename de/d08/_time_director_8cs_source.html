108 public void BulletTime(float peakTime, float duration, Ease toPeakEaseType = Ease.OutExpo, Ease toResumeEaseType = Ease.Linear, Action callback = null)
109 {
110 if (IsPaused) return;
111 if (HitStopTween != null && HitStopTween.IsActive())
112 {
113 RemoveHitStop();
114 }
115
116 BulletTimeTween?.Kill();
117
118 var effectDuration = duration * 0.9f * peakTime;
119 var restDuration = duration * peakTime - effectDuration;
120
121 PreviousTimeScale = Time.timeScale;
122 BulletTimeTween = DOTween.Sequence()
123 .Append(DOTween.To(() => Time.timeScale, x => Time.timeScale = x, peakTime, effectDuration)
124 .SetEase(toPeakEaseType)
125 .SetUpdate(DG.Tweening.UpdateType.Fixed, true)
126 .OnUpdate(() => Time.fixedDeltaTime = Time.timeScale * 0.02f))
127 .Append(DOTween.To(() => Time.timeScale, x => Time.timeScale = x, 1f, restDuration)
128 .SetEase(toResumeEaseType)
129 .SetUpdate(DG.Tweening.UpdateType.Fixed, true) // Ignore time scale
130 .OnUpdate(() => Time.fixedDeltaTime = Time.timeScale * 0.02f))
131 .OnKill(() => BulletTimeTween = null)
132 .OnComplete(() =>
