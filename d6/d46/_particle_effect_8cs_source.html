79 protected override void PlayEffect()
80 {
81 if (pool.IsSingleInstance)
82 {
83 pool.ReleaseAll();
84 }
85
86 if (spawnParent == SpawnParent.Target && TargetObject.IsNullOrDestroyed())
87 {
88 Stop();
89 return;
90 }
91
92 if (spawnParent == SpawnParent.Provider && ProvideObject.IsNullOrDestroyed())
93 {
94 Stop();
95 return;
96 }
97
98 // Spawn Parent
99 var parent = spawnParent switch
100 {
101 SpawnParent.ChildOfThis => transform,
102 SpawnParent.Provider => ProvideObject.transform,
103 SpawnParent.Target => TargetObject.transform,
104 SpawnParent.ProviderPreTransform => GetPrePosition(ProvideObject, preTransformType),
105 SpawnParent.TargetPreTransform => GetPrePosition(TargetObject, preTransformType),
106 SpawnParent.Custom => customSpawnTransform,
107 SpawnParent.World => null,
108 _ => null
109 };
110
111 // SpawnPosition
112 var worldPosition = spawnPosition switch
113 {
114 SpawnPosition.InheritParent => parent.IsNullOrDestroyed() ? transform.position : parent!.position,
115 SpawnPosition.ProviderPreposition => GetPrePosition(ProvideObject, prePosition).position,
116 SpawnPosition.TargetPreposition => GetPrePosition(TargetObject, prePosition).position,
117 _ => transform.position,
118 };
119
120 // SpawnRotation
121 var particle = pool.Get();
122 var particleRotation = particle.transform.rotation;
123 var providerTransform = ProvideObject.IsNullOrDestroyed() ? transform : ProvideObject.transform;
124 var targetTransform = TargetObject.IsNullOrDestroyed() ? transform : TargetObject.transform;
125 var providerPosition = providerTransform.position;
126 var targetPosition = targetTransform.position;
127 var worldRotation = spawnRotation switch
128 {
129 SpawnRotation.None => particleRotation,
130 SpawnRotation.ParticleSelf => pool.Prefab.transform.rotation,
131 SpawnRotation.ProvideObject => providerTransform.rotation,
132 SpawnRotation.TargetObject => targetTransform.rotation,
133 SpawnRotation.ProviderToTarget => Quaternion.LookRotation(targetPosition - providerPosition),
134 SpawnRotation.TargetToProvider => Vector3.Distance(providerPosition, targetPosition) == 0f
135 ? Quaternion.identity
136 : Quaternion.LookRotation(providerPosition - targetPosition),
137 _ => particleRotation,
138 };
139
140 // Offset
141 var worldOffset = transform.TransformDirection(offset);
142 var destination = worldPosition + worldOffset;
143
144 // Play
145 particle.transform.SetPositionAndRotation(destination, worldRotation);
146 particle.transform.SetParent(parent, true);
147 particle.Play();
148 }
