37 public void GenerateMapFromMatrix()
38 {

40 if (CurrentMap.IsNullOrDestroyed()) return;
41
42 DeleteAll();
43 CurrentMap.PrePositionTableInitialize();
44
45 var offsetX = -matrix.First().Count / 2;
46 var offsetZ = -matrix.Count / 2;
47 var z = 0;
48
49 for (var i = matrix.Count - 1; 0 <= i; i--)
50 {
51 var x = 0;
52 var currRow = matrix[i];
53
54 for (var j = 0; j < currRow.Count; j++)
55 {
56 var pos = Grid.GetCellCenterWorld(new Vector3Int(x + offsetX, z + offsetZ, 0));
57 var key = currRow[j].Trim().ToLower();
58
59 switch (key)
60 {
61 case "w":
62 {
63 CreatePrefab(WallPrefab, pos, WallContainer);
64
65 if (generateDummyWalls)
66 {
67 CreatePrefab(WallDummyPrefab, pos, WallDummyContainer);
68 }
69 break;
70 }
71 case "v":
72 {
73 CreatePrefab(VoidPrefab, pos, VoidContainer);
74 break;
75 }
76 case "t1":
77 {
78 CreatePrefab(FloorTrapPrefabs.Random(), pos, EnvironmentContainer);
79 break;
80 }
81 case "t2":
82 {
83 CreatePrefab(VerticalTrapPrefabs.Random(), pos, EnvironmentContainer);
84
85 var obstacle = CreatePrefab(ObstaclePrefabs.Random(), pos, EnvironmentContainer);
86 obstacle.transform.localRotation = Quaternion.Euler(new Vector3(0, Random.value * 360f, 0));
87 break;
88 }
89 case "s":
90 {
91 CreatePrefab(SpiritJarPrefabs.Random(), pos, EnvironmentContainer);
92 break;
93 }
94 case "o":
95 {
96 var obstacleTransform = CreatePrefab(ObstaclePrefabs.Random(), pos, EnvironmentContainer).transform;
97
98 obstacleTransform.localRotation = rotateObstacle

100 : obstacleTransform.localRotation;
101
102 break;
103 }
104 case "sp":
105 {
106 CreatePrefab(FiendSpawnPin, pos, PrePositionTable[MapPrepositionType.FiendSpawns]);
107 break;
108 }
109 case "e1":
110 {
111 PrePositionTable[MapPrepositionType.PlayerSpawn].position = pos;
112

119 break;
